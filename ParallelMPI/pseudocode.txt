  int div = rank/root;
  int mod = rank%root;
  xLeft = xLeft + div*(length/root);
  xRight = xLeft + (div+1)*(length/root)
  yBottom = yBottom + mod*(length/root);
  yUp = yBottom + (mod+1)*(length/root);
